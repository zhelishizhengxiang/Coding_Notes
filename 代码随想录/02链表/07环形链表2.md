public class Solution {
    //x=(n-1)(y+z)+z
    public ListNode detectCycle(ListNode head) {
        ListNode fast=head;
        ListNode slow=head;
        while(fast!=null && fast.next!=null){
            slow=slow.next;
            fast=fast.next.next;
            //说明链表有环
            if(slow==fast){
                ListNode index=head;
                ListNode res=slow;
                while(index!=res){
                    index=index.next;
                    res=res.next;
                }
                return index;
            }
        }
        return null;

    }
}
