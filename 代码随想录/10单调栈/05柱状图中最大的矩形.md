[84. 柱状图中最大的矩形 - 力扣（LeetCode）](https://leetcode.cn/problems/largest-rectangle-in-histogram/description/)
```java
class Solution {
    public int largestRectangleArea(int[] heights) {
        //此题和接雨水完全相反：使用单调递减栈来找右边比他小的元素，左边比他小的元素，就是栈中的第二个元素

        //但是需要在该数组中前面和后面都加上0，因为避免（1,2,3,4）这种数据这种一直加入栈但是不处理的情况;以及（4.3.2.1）加入栈时，栈中元素只有一个，没有第二个元素的情况
        int[] nums=new int[heights.length+2];
        for(int i=0;i<heights.length;i++){
            nums[1+i]=heights[i];
        }
        Stack<Integer> stack=new Stack<>();
        stack.push(0);
        int result=0;
        for(int i=1;i<nums.length;i++){
            //递减栈如果即将加入的元素大于栈顶元素，那么就直接加入
            if(nums[i]>nums[stack.peek()])
                stack.push(i);
            //和接雨水一样，可以不管也可以弹出在入
            else if(nums[i]==nums[stack.peek()]){
                stack.pop();
                stack.push(i);
            }
            //如果当前元素小于站定元素,
            else{
                while(!stack.isEmpty() && nums[i]<nums[stack.peek()]){
                    int mid=stack.pop();
                    //一定要判断是否里面还有元素，因为进行的出栈
                    if(!stack.isEmpty()){
                        int h=nums[mid];
                        int w=i-stack.peek()-1;
                        result=Math.max(result,h*w);
                    }
                }
                stack.push(i);
            }

        }
        return result;
        
    }
}
```
