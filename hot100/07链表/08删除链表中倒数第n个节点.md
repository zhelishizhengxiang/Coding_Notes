19. 删除链表的倒数第 N 个结点 - 力扣（LeetCode）](https://leetcode.cn/problems/remove-nth-node-from-end-of-list/description/)
```java
class Solution {
    public ListNode removeNthFromEnd(ListNode head, int n) {
        //删掉导数第n个结点，就要找到倒数第n+1个结点
        n++;
        //使用虚拟节点统一操作,防止删除的是丢给结点
        ListNode shead=new ListNode(0,head);
        ListNode fast=shead;
        ListNode slow=shead;
        //先让一个节点走n+1个，然后一起走，当快指针为空时，慢指针指的就是倒数第n+1个节点
        while(n-->0 && fast!=null){
            fast=fast.next;
        }

        while(fast!=null){
            fast=fast.next;
            slow=slow.next;
        }

        //此时找到要删除节点的前一个节点
        slow.next=slow.next.next;
        return shead.next;
    }
}
```

```java
class Node{

public class Main {
    public static Node func(Node head,int n){
        Node shead=new Node(-1);
        shead.next=head;
        n++;
        Node fast=shead;
        Node slow=shead;
        while(n-->0 && fast!=null){
            fast=fast.next;
        }
        while(fast!=null){
            fast=fast.next;
            slow=slow.next;
        }
        slow.next=slow.next.next;
        return shead.next;
    }
    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);

        int num=scanner.nextInt();
        Node head=new Node(scanner.nextInt());
        Node tmp=head;
        for(int i=1;i<num;i++){
            Node node=new Node(scanner.nextInt());
            tmp.next=node;
            tmp=node;
        }
        int n=scanner.nextInt();

        head=func(head,n);
        if(head==null)
            System.out.println();

            tmp=head;
            while(tmp.next!=null){
                System.out.print(tmp.val+" ");
                tmp=tmp.next;
            }

        }
        
    }
```
