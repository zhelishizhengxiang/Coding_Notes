[25. K 个一组翻转链表 - 力扣（LeetCode）](https://leetcode.cn/problems/reverse-nodes-in-k-group/description/?envType=study-plan-v2&envId=top-100-liked)

两种版本处理返回的方式各有异同，使用方式二的方法会好一点，在反转链表的时候就把指针确定好

主要几个注意点：
1. reverse之后一定要移动指针，使得end指向当前遍历过的地方，如果不变的话就相当于移动一个节点就会反转链表一次
2. reverse传入的一定是传入要翻转链表的区段的前一个节点,和后面一个节点
```java
class Solution {
    //传入要翻转链表的区段的前一个节点,和后面一个节点
    public void reverseList(ListNode begin,ListNode end){
        //让第一个节点指向end
        ListNode pre=end;
        ListNode cur=begin.next;
        while(cur!=end){
            ListNode tmp=cur.next;
            cur.next=pre;
            pre=cur;
            cur=tmp;
        }
        //让begin指向反转之后的头结点
        begin.next=pre;
    }
    public ListNode reverseKGroup(ListNode head, int k) {
        ListNode shead=new ListNode(0,head);
        ListNode begin=shead;//记录当前区段的前一个位置
        ListNode end=head;//从第一个节点开始，这样翻转的节点就在beign和end之间
        //记录了当前区间的第一个节点
        ListNode pre=null;
    
        while(end!=null){
            // 检查剩余节点是否足够k个
            for(int i=0;i<k;i++){
                if(end!=null){
                    if(i==0)
                        pre=end;//记录了当前区间的第一个节点
                    end=end.next;
                }
                else
                    return shead.next;
            }
             // 反转当前k个节点
            reverseList(begin,end);
            // 反转后begin应该指向反转前的第一个节点
            begin=pre;
           
        }
        return shead.next;
    }
}
```

```java
import java.util.*;


class Node{
    int val;
    Node next;
    public Node(int val){
        this.val=val;
    }
}

public class Main{
    public static Node reverse(Node begin,Node end){
        Node slow=end;
        Node fast=begin.next;
        Node cur=begin.next;
        while(fast!=end){
            Node tmp=fast.next;
            fast.next=slow;
            slow=fast;
            fast=tmp;
        }
        begin.next=slow;
        return cur;
    }
    public static Node func(Node head,int k){
        Node shead=new Node(-1);
        shead.next=head;
        Node pre=shead;
        Node tmp=shead;
        int count=0;
        while(tmp!=null){
            if(count==k){
                tmp=reverse(pre,tmp.next);
                pre=tmp;
                count=0;
            }
            tmp=tmp.next;
            count++;
            
        }
        return shead.next;
    }
    public static void main(String[] args){
        Scanner scanner=new Scanner(System.in);

        Node head=new Node(scanner.nextInt());
        Node tmp=head;
        Node pre=head;
        while(scanner.hasNext()){
            Node node=new Node(scanner.nextInt());
            tmp.next=node;
            pre=tmp;
            tmp=node;
        }
        int k=tmp.val;
        pre.next=null;
        head=func(head,k);

        tmp=head;
        while(tmp.next!=null){
            System.out.print(tmp.val+" ");
            tmp=tmp.next;
        }
        System.out.println(tmp.val);

        
    }
}
```
