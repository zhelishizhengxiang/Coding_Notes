```java
class Solution {
    //在递归的过程中，直径肯定在某个递归栈中执行该程序时会经过节点跟节点
    //使用一个全局变量来维护直径的最大值，
    int res=0;
    //递归做了两件事：
    // 1.计算以root的根节点的直径长度
    // 2.向上一层返回root节点的最大深度（知道两个子节点的最大深度，才可以知道经过root的任意两节点的最长路径的长度（并且两个节点肯定是叶子节点））
    public int traversal(TreeNode root){
         //因为求的是边的个数，所以空节点的边为-1，叶子节点的直径长度的0
        if(root==null)
            return -1;
        int left=traversal(root.left);
        int right=traversal(root.right);
        //1.计算以root的根节点的直径长度
        res=Math.max(res,left+right+2);

        //返回上一层root节点的最大深度
        return Math.max(left,right)+1;
    }
    public int diameterOfBinaryTree(TreeNode root) {
       traversal(root);
       return res;
    }
}
```

```java
import java.util.*;

class TreeNode{
    int val;
    TreeNode left;
    TreeNode right;
    public TreeNode(int val){
        this.val=val;
    }
}

public class Main {
    public static TreeNode build(String[] nums){
        if(nums.length==0)
            return null;
        Queue<TreeNode> queue=new ArrayDeque<>();
        TreeNode root=new TreeNode(Integer.parseInt(nums[0]));
        queue.offer(root);
        int index=1;
        while(index<nums.length && !queue.isEmpty()){
            TreeNode node=queue.poll();

            if(index<nums.length && !nums[index].equals("null")){
                TreeNode newnode=new TreeNode(Integer.parseInt(nums[index]));
                node.left=newnode;
                queue.offer(newnode);
            }
            index++;

            if(index<nums.length && !nums[index].equals("null")){
                TreeNode newnode=new TreeNode(Integer.parseInt(nums[index]));
                node.right=newnode;
                queue.offer(newnode);
            }
            index++;
        }
        return root;
    }
    static int res=0;
    public static int func(TreeNode root){
        if(root==null)
            return -1;
        int left=func(root.left);
        int right=func(root.right);
        res=Math.max(res,left+right+2);
        return 1+Math.max(left,right);
        
    }
    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);

        String[] nums=scanner.nextLine().split(" ");
        TreeNode root=build(nums);
        if(root==null){
            System.out.println("true");
            return;
        }
        func(root);
        System.out.println(res);
    }
}
```
