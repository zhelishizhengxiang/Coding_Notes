与[[01不同路径]]一模一样，只是递推公式发生了一点变化，求路径个数变成了最小路径和

```java
class Solution {
    public int minPathSum(int[][] grid) {
        //dp含义:到达(i,j)点的最小路径和为dp[i][j]，
        int[][] dp=new int[grid.length][grid[0].length];
        //初始化行
        int sum=0;
        for(int i=0;i<grid[0].length;i++){
            sum+=grid[0][i];
            dp[0][i]=sum;
        }
        //初始化列
        sum=0;
        for(int i=0;i<grid.length;i++){
            sum+=grid[i][0];
            dp[i][0]=sum;
        }

        for(int i=1;i<grid.length;i++){
            for(int j=1;j<grid[0].length;j++){
                //元素对应的最小路径和等于其上方相邻元素与其左方相邻元素两者对应的最小路径和中的最小值加上当前元素的值
                dp[i][j]=Math.min(dp[i-1][j],dp[i][j-1])+grid[i][j];
            }
        }
        return dp[grid.length-1][grid[0].length-1];
    }
}
```

```java
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Main {
    public static  int func(List<List<Integer>> list) {
        //以下标i最长的有效括号子串的长度为dp[i]
        int[][] dp=new int[list.size()][list.get(0).size()];
        int sum=0;
        for(int i=0;i<list.size();i++){
           sum+=list.get(i).get(0);
           dp[i][0]=sum;
        }
        sum=0;
        for(int i=0;i<list.get(0).size();i++){
            sum+=list.get(0).get(i);
            dp[0][i]=sum;
        }

        for (int i = 1; i < list.size(); i++) {
            for (int j = 1; j < list.get(i).size(); j++) {
                dp[i][j]=Math.min(dp[i-1][j],dp[i][j-1])+list.get(i).get(j);
            }
        }
        return dp[list.size()-1][list.get(0).size()-1];
    }

    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        List<List<Integer>> graph=new ArrayList<>();
        while(sc.hasNextLine()){
           String str=sc.nextLine();
           if(str.isEmpty())
               break;
           String[] strs=str.split(" ");
            ArrayList<Integer> list = new ArrayList<>();
            for(int i=0;i<strs.length;i++){
                list.add(Integer.parseInt(strs[i]));
            }
            graph.add(list);

        }
        System.out.println(func(graph));



    }
}
```
